my_logpost <- function(unknowns, my_info)
{
  # unpack the unknowns into separate variables
  lik_mu <- unknowns[1]
  lik_sigma <- unknowns[2]
  
  # calculate the log-likelihood
  log_lik <- 8*log(1/(sqrt(2*pi*lik_sigma^2)))+ (-1/(2*lik_sigma^2))*sum((my_info$xobs-lik_mu)^2)
  
  # calculate the log-prior on mu
  log_prior_mu <- log(1/(sqrt(2*pi*my_info$tau_0^2)))+ (-1/(2*my_info$tau_0^2))*(lik_mu-my_info$mu_0)^2
  
  # calculate the log-prior on sigma
  log_prior_sigma <- log(my_info$sigma_rate) - my_info$sigma_rate *lik_sigma
  
  # return the (un-normalized) log-posterior
  return(log_lik+log_prior_mu+log_prior_sigma)
  
}
